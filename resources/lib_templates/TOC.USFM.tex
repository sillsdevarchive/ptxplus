% Macro code for leadered two colum layout:
\newdimen\tocheaderlabel
\def\tocheader#1#2{\parfillskip=\tblmarginright\tah #1\tah*\hfil
  \setbox0=\hbox{\tah #2\tah*}\tocheaderlabel=\wd0\box0\par\vskip \tblrowspace}
\def\tocrow#1#2{\parskip=\tblheaderspace\parfillskip=\tblmarginright\noindent\tar #1\tar*%
  \quad\leaders\hbox to \leaderspace{\hss\myleader\hss}\hfill\hbox to \tocheaderlabel{%
  \hfil\tar #2\tar*}\par%
\baselineskip=16pt}
